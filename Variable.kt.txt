fun main(){
val myNum: Int = 5                // Int
val myDoubleNum: Double = 5.99    // Double
val myLetter: Char = 'D'          // Char
val myBoolean: Boolean = true     // Boolean
val myText: String = "Hello"      // String
println(myNum)
println(myDoubleNum)
println(myLetter)
println(myBoolean)
println(myText)
println(myText.lowercase())
println(myText+" "+"World")
println(myText.length)
println(myText.plus("World"))
println(myText[1])
println("This is $myText")
}
